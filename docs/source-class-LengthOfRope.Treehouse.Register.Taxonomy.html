  5: /**
  6:  * A helper class to create Custom Taxonomies.
  7:  *
  8:  * It handles:
  9:  * - Creation of the taxonomy (with a prefix by default)
 10:  * - Adding templates within the plugin for singles and archives (using setupTemplates method).
 11:  * - Allows basic settings to be set through chaining
 12:  *
 13:  * Note: To make sure all translations are ready this should be called in an init action!
 14:  *
 15:  * Usage (ordinary mode):
 16:  * <code>
 17:  * new \LengthOfRope\Treehouse\Register\Taxonomy('product_categories', array(
 18:  *      'labels'      => array(
 19:  *          'name'          => __('Products'),
 20:  *          'singular_name' => __('Product')
 21:  *      ),
 22:  *      'public'      => true,
 23:  *      'has_archive' => true,
 24:  *  ));
 25:  * </code>
 26:  *
 27:  * Usage (chained mode):
 28:  * <code>
 29:  * \LengthOfRope\Treehouse\Register\Taxonomy::factory('product_categories')
 30:  *      ->setLabels(__('Product categories'), __('Product category'))
 31:  *      ->setPublic(true)
 32:  *      ->setHasArchive(true);
 33:  * </code>

 47:      *
 48:      * @param string $slug The unique slug for this taxonomy.
 49:      * @param string|array $objectType Name of the object type for the taxonomy object.
 50:      *                                 Object-types can be built-in Post Type or any Custom Post Type
 51:      *                                 that may be registered. If null, the taxonomy must be manually added to the
 52:      *                                 custom post type where it should be available.
 53:      *                                 Note: remember the slugPrefix the Treehouse classes are adding!
 54:      * @param array $args
 55:      * @param string $slugPrefix
 56:      */
 57:     public function __construct($slug, $objectType = null, $args = array(), $slugPrefix = 'th_tax_')
 58:     {
 59:         // Add the slugPrefix to make sure it is a unique posttype.
 60:         $this->slug = $slugPrefix . $slug;
 61: 
 62:         $this->objectType = $objectType;
 63: 
 64:         // Make sure the 'slug' on the frontend does not have the prefix
 65:         if (is_array($args) && $slugPrefix !== '' && !isset($args['rewrite'])) {
 66:             $args['rewrite'] = array(
 67:                 'slug' => $slug
 68:             );
 69:         }
 70: 
 71:         // Set some default values so we don't have to check for them in all set methods
 72:         $defaults   = array(
 73:             'labels' => array(),
 74:         );
 75:         $this->args = array_merge($defaults, $args);
 76: 
 77:         // Register the post type late in the init process, since it is probably created in a init action as well.
 78:         add_action('init', function() {
 79:             $this->registerTaxonomy();
 80:         }, 99);
 81:     }

313:     /**
314:      * A function name that will be called when the count of an associated $object_type, such as post, is updated.
315:      * Works much like a hook.
316:      * Default: None - but see Note, below.
317:      *
318:      * Note: While the default is '', when actually performing the count update in wp_update_term_count_now(), if
319:      * the taxonomy is only attached to post types (as opposed to other WordPress objects, like user),
320:      * the built-in _update_post_term_count() function will be used to count only published posts associated
321:      * with that term, otherwise _update_generic_term_count() will be used instead, that does no such checking.
322:      *
323:      * This is significant in the case of attachments. Because an attachment is a type of post, the
324:      * default _update_post_term_count() will be used. However, this may be undesirable, because this will
325:      * only count attachments that are actually attached to another post (like when you insert an image into
326:      * a post). This means that attachments that you simply upload to WordPress using the Media Library, but
327:      * do not actually attach to another post will not be counted. If your intention behind associating a
328:      * taxonomy with attachments was to leverage the Media Library as a sort of Document Management solution,
329:      * you are probably more interested in the counts of unattached Media items, than in those attached to posts.
330:      * In this case, you should force the use of _update_generic_term_count() by setting '_update_generic_term_count'
331:      * as the value for update_count_callback.
332:      *
333:      * Another important consideration is that _update_post_term_count() only counts published posts. If you are
334:      * using custom statuses, or using custom post types where being published is not necessarily a consideration
335:      * for being counted in the term count, then you will need to provide your own callback that doesn't include
336:      * the post_status portion of the where clause.
337:      *

339:      * @return \LengthOfRope\Treehouse\Register\Taxonomy
